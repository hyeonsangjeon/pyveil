from pyveil import Channel, Veil

veil = Veil.high(
    secret=b"tenant-secret",
    scope="tenant/session",
)

safe = veil.redact_data(
    messages,
    channel=Channel.PROMPT_INPUT,
)

response = call_llm(safe.data)
